function highAndLow(numbers) {
  let numbersArray = numbers.split(' ').map(x=>+x);
  let max = numbersArray.reduce(function(a, b) {
    return Math.max(a, b);
  });
  let min = numbersArray.reduce(function(a, b) {
    return Math.min(a, b);
  });
  return max.toString() + ' ' + min.toString();
}
